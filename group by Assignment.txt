Problem 1:
SELECT sector, round(avg(Eps),2) as AvgEPSPerSector 
from Stock group by sector having AvgEPSPerSector>10 
order by AvgEPSPerSector desc;


Problem 2:
SELECT
  job_id,
  AVG(salary) AS avg_salary
FROM
  employee
GROUP BY
  job_id
HAVING
avg_salary>10000
ORDER BY
  avg_salary DESC;

Problem 3:
SELECT
  job_id,
  AVG(salary) AS avg_salary
FROM
  employee
GROUP BY
  job_id
 HAVING avg_salary>10000 AND avg_salary<20000
ORDER BY
  avg_salary DESC;

Problem 4:
SELECT product_id,COUNT(*) 
FROM sale GROUP BY product_id 
HAVING COUNT(*) 
ORDER BY COUNT(*) DESC;
